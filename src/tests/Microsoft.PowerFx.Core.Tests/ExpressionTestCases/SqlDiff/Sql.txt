// math operation overflow
>> Power(10,12)
/*SqlRunner*/ #Error
1000000000000

>> Power(100,6)
/*SqlRunner*/ #Error
1000000000000

>> Power(10,200)
/*SqlRunner*/ #Error
1E+200

>> IsError(Power(10,200))
/*SqlRunner*/ true
false

>> RoundUp(1, -200)
/*SqlRunner*/ #Error
1E+200

>> RoundUp(1, -10)
/*SqlRunner*/ #Error
10000000000

>> Text(Ln(Exp(200)))
/*SqlRunner*/ #Error
"200"

>> IsError(Text(Ln(Exp(200))))
/*SqlRunner*/ true
false

// literal overflow
>> 1E37
/*SqlRunner*/ #Error
1E+37

>> IsError(1E37)
/*SqlRunner*/ true
false
